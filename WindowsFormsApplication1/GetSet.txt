프로퍼티(Property) 기본형식

class 클래스명
{
    데이터 타입 필드명;
    접근한정자 데이터타입 프로퍼티명
    {
        get		//getField메소드를 대체. 속성값을 반환하는데 사용됨//읽기
        {
	return 필드명;	
        }
        set			//setField메소드를 대체. 새값 할당//쓰기
        {
	필드명 = value;
        }
    }
}
